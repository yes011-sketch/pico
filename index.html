import network import urequests import time from machine import Pin, PWM # WiFi 설정 ssid = "YOUR_WIFI" password = "YOUR_PASSWORD" # Firebase URL url = "https://pico-d188e-default-rtdb.asia-southeast1.firebasedatabase.app/.json" # WiFi 연결 wlan = network.WLAN(network.STA_IF) wlan.active(True) wlan.connect(ssid, password) print("WiFi 연결 중...") while not wlan.isconnected(): time.sleep(1) print("WiFi 연결 완료:", wlan.ifconfig()) # PWM 설정 (0~65535) led_r = PWM(Pin(15)) led_g = PWM(Pin(14)) led_b = PWM(Pin(13)) led_r.freq(1000) led_g.freq(1000) led_b.freq(1000) def set_color(r, g, b): # 0~255 → 0~65535 변환 led_r.duty_u16(int(r * 257)) led_g.duty_u16(int(g * 257)) led_b.duty_u16(int(b * 257)) while True: try: res = urequests.get(url) data = res.json() res.close() r = data.get("Red", 0) g = data.get("Green", 0) b = data.get("Blue", 0) print("RGB:", r, g, b) set_color(r, g, b) except Exception as e: print("에러:", e) time.sleep(1)
